select
NTPE_C_NROADMIN  	CIP,
NTFA_C_NROADFAM  	CIP_FAM,
CASE WHEN SUBSTR (NTFA_C_NROADFAM, '8', '1') = '2' AND NTFA_C_SEXO = 'F' THEN 'ESPOSA'
     WHEN SUBSTR (NTFA_C_NROADFAM, '8', '1') = '2' AND NTFA_C_SEXO = 'M' THEN 'ESPOSO'
     WHEN SUBSTR (NTFA_C_NROADFAM, '8', '1')= '4' AND NTFA_C_SEXO = 'F' THEN 'MADRE'
     WHEN SUBSTR (NTFA_C_NROADFAM, '8', '1')= '3' AND NTFA_C_SEXO = 'M' THEN 'PADRE'
     WHEN SUBSTR (NTFA_C_NROADFAM, '8', '2') BETWEEN '50'AND'70' AND NTFA_C_SEXO = 'F' THEN 'HIJA'
     WHEN SUBSTR (NTFA_C_NROADFAM, '8', '2') BETWEEN '50'AND'70' AND NTFA_C_SEXO = 'M' THEN 'HIJO'
END CASE,  
NTFA_C_NOMAPE  	NOMBRE_FAM,
NTFA_D_FECHNAC		FECHNAC,
NTFA_C_TIPODOCU		TIPODOCU,
NTFA_C_NRODOCU		NRODOCU,
NTFA_C_SEXO  	SEXO_FAM
from
copere.nm_persona,
copere.nm_militares,
copere.nt_grados,
copere.nt_armas,
copere.NT_SITUAMIL,
copere.NM_FAMILIA
where	
	NTPE_C_NROADMIN 	= NTMI_C_NROADMIN
AND 	NTGR_C_CODIGO		= NTPE_C_GRADO
AND 	NTAR_C_CODIGO		= NTMI_C_ARMA 
AND     NTSM_C_CODIGO		= NTMI_C_SITUAMIL 
AND	NTPE_C_NROADMIN		= NTFA_C_NROADMIN
AND     NTMI_C_SITUAMIL		IN ('1','2')
AND	NTPE_C_TIPOPERS		IN ('A','B')
ORDER BY NTGR_C_CODIGO,ntar_c_codigo, ntpe_c_apellnom;
